Class
package examples;
public class HelloMain {
    public static void main(String[] params) {
        System.out.println("Hello world:" + params[0]);
    }
}

build.gradle
apply plugin:'java'
apply plugin:'application'

mainClassName='examples.HelloMain'

run {
    args username
}
