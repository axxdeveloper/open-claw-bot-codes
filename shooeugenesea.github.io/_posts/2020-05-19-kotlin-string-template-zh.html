  • until
  • downTo
  • 用字元比較
  • 轉換成 list
package examples

fun main() {
    val age = 100
    if (age in 150 downTo 0) {
        println("downTo")
    }
    if (age in 0 until 100) {
        println("until")
    }
    val c = 'X'
    val s = c in 'a'..'q'
    println(s)


    val downToList = (10 downTo 0).toList()
    println(downToList)


    val untilList = (0 until 10).toList()
    println(untilList)


    val b = false
    val ss = "${if(b) "Yes" else "No"}"
    println(ss)
}
