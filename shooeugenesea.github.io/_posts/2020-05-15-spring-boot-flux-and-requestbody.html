  • Controller
    • @RestController
    public class RequestBodyController {

        @PostMapping("/requestBodyFlux")
        public Mono<Void> postFlux(@RequestBody Flux<String> ids) {
            return ids.map(id -> {
                String s = "id:" + id;
                System.out.println(s);
                return s;
            }).then();
        }

    }
