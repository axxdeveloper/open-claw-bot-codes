When
package examples

fun main() {
    var name = "John"
    val greet = when(name) {
        "John" -> "Hello John"
        else -> "Unknown"
    }
    println(greet)
}

When + if else
package examples

fun main() {
    val i = 50
    val s = when(i) {
        in 0..50 -> "0-50"
        in 51..100 -> "51-100"
        else -> if (i in 101..150) "101-150" else "Out Of Range"
    }
}
