定義變數
package examples

fun main(args: Array<String>) {
    // variable type declaration
    var i = 0;
    var i2: Int = 0
    println("hwllo " + ++i + ++i2)

    // can skip type declaration
    var s = "string"
    println("string: " + s)

    var c : Char = 'a'
    println("Char" + c)

    var b = true
    var d = 0.123
    var list = listOf("a","b","c")
    println(list)

    var set : Set<String> = setOf("d","e", "f", "d")
    println(set)

    var map = mapOf("a" to 1, "b" to 2)
    println(map)

    // read only => val
    val pp = i
    println(pp)
}
